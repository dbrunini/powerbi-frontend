export default function handler(req, res) { // Adicionando cabeçalhos de CORS res.setHeader("Access-Control-Allow-Origin", "https://dbrunini.github.io"); // Substitua "SEU_USUARIO" pelo nome do seu usuário do GitHub res.setHeader("Access-Control-Allow-Methods", "POST, OPTIONS"); res.setHeader("Access-Control-Allow-Headers", "Content-Type"); // Resolvendo solicitações de preflight CORS if (req.method === "OPTIONS") { res.status(200).end(); return; } // Processo de autenticação normal const { username, password } = req.body; const validUser = process.env.USERNAME; const validPassword = process.env.PASSWORD; if (username === validUser && password === validPassword) { res.status(200).json({ success: true, link: process.env.POWER_BI_LINK }); } else { res.status(401).json({ success: false }); } }
